#include<stdio.h>
#include<fcntl.h>
#include<errno.h>
#include<unistd.h>
#include<string.h>
#include<sys/types.h>
#include<pwd.h>
#include<grp.h>
#include<stdlib.h>
int main(){
int fd1,fd2;
int sz;
char c;
fd1=open("empty.c",O_RDONLY|O_CREAT,0777);
fd2=open("mi.c",O_WRONLY|O_CREAT,0777);
sz=lseek(fd1,-1,SEEK_END);
for(int i=0;i<sz;i++){
lseek(fd1,i,SEEK_SET);
lseek(fd2,i,SEEK_SET);
read(fd1,&c,1);
write(fd2,&c,1);
printf("%c ",c);
}
printf("\n");
return 0;
}
